using namespace std;
int demdigit(int a)
{
	int dem = 0;
	while (a != 0)
	{
		dem++;
		a /= 10;
	}
	return dem;
}
bool palindrome(int num)
{
	int r, sum = 0, temp;
	for (temp = num; num != 0; num = num / 10) {
		r = num % 10;
		sum = sum * 10 + r;
	}
	if (temp == sum)
		return 1;
	else
		return 0;
}
int main()
{
	int t;
	cin >> t;
	//cout << demdigit(t);
	while (t--)
	{
		int x;
		cin >> x;
		int du = x % 10;
		int dem = (du - 1) * 10;
		int count = demdigit(x);
		if (count == 1)
			dem += 1;
		else if (count == 2)
			dem += 3;
		else if (count == 3)
			dem += 6;
		else if (count == 4)
			dem += 10;
		cout << dem << endl;
	}
}
